def _allocate_loopback_port() -> int:
    """Bind to port 0 on 127.0.0.1 to get a free port, then release it."""
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as sock:
        sock.bind(("127.0.0.1", 0))
        port = sock.getsockname()[1]
    return port


def _wait_for_http_ready(port: int, timeout_s: float = 10.0) -> None:
    """Poll the loopback port until something is accepting connections.

    Mirrors the Phase 53 HTTP-02 socket-bind verification pattern.
    Fails the test if the listener doesn't come up within timeout_s.
    """
    deadline = time.monotonic() + timeout_s
    last_exc: Exception | None = None
    while time.monotonic() < deadline:
        try:
            with socket.create_connection(("127.0.0.1", port), timeout=0.5):
                return
        except (ConnectionRefusedError, OSError) as exc:
            last_exc = exc
            time.sleep(0.1)
    raise RuntimeError(
        f"agent-brain-mcp HTTP listener never came up on 127.0.0.1:{port} "
        f"within {timeout_s}s; last error: {last_exc!r}"
    )

# -----------------------------------------------------------------------
# Shared chunk extractor — tolerates SDK-shape drift across versions
# -----------------------------------------------------------------------

def _extract_chunks(result: Any) -> list[Any]:
    """Pull the chunk list out of an MCP call_tool response.

    The shape may be:
    - dict with key 'chunks' (preferred, current agent-brain-mcp surface)
    - dict with key 'results' (alternate naming)
    - list (raw)
    - object with .content attribute carrying JSON-encoded text

    Returns [] if no chunks found — caller asserts len(chunks) > 0.
    """
    if result is None:
        return []
    if isinstance(result, list):
        return result
    if isinstance(result, dict):
        for key in ("chunks", "results", "documents"):
            if key in result and isinstance(result[key], list):
                return result[key]
    # MCP SDK may wrap the tool output in a CallToolResult with .content.
    content = getattr(result, "content", None)
    if content:
        # content is typically a list[TextContent]; .text holds JSON.
        import json

        for item in content:
            text = getattr(item, "text", None)
            if not text:
                continue
            try:
                parsed = json.loads(text)
            except (ValueError, TypeError):
                continue
            if isinstance(parsed, list):
                return parsed
            if isinstance(parsed, dict):
                for key in ("chunks", "results", "documents"):
                    if key in parsed and isinstance(parsed[key], list):
                        return parsed[key]
    return []
