```python
"""FRAME-03: LlamaIndex adapter smoke test (stdio transport).

from __future__ import annotations

import asyncio
import os
import shutil
from typing import Any

import pytest

# Module-level optional import.
try:
    from llama_index.tools.mcp import (  # type: ignore[import-not-found]
        BasicMCPClient,
        McpToolSpec,
    )

    HAS_LLAMA_INDEX_MCP = True
except ImportError:
    BasicMCPClient = None  # type: ignore[assignment,misc]
    McpToolSpec = None  # type: ignore[assignment,misc]
    HAS_LLAMA_INDEX_MCP = False


# Phase 60 hygiene helper hoisted to Plan 61-01 conftest. Importing the
# shared definition keeps the allowlist drift to a single file.
from tests.framework.conftest import filtered_subprocess_env  # noqa: E402


def _extract_chunks(result: Any) -> list[Any]:
    """Defensive shape extractor — mirrors Plan 61-02/03 helper."""
    if result is None:
        return []
    if isinstance(result, list):
        return result
    if isinstance(result, str):
        import json

        try:
            parsed = json.loads(result)
        except (ValueError, TypeError):
            return []
        if isinstance(parsed, list):
            return parsed
        if isinstance(parsed, dict):
            for key in ("chunks", "results", "documents"):
                if key in parsed and isinstance(parsed[key], list):
                    return parsed[key]
        return []
    if isinstance(result, dict):
        for key in ("chunks", "results", "documents"):
            if key in result and isinstance(result[key], list):
                return result[key]
    # LlamaIndex tool results may have .raw_output or .content attrs.
    raw = getattr(result, "raw_output", None) or getattr(result, "content", None)
    if raw is not None:
        return _extract_chunks(raw)
    return []


@pytest.mark.framework_matrix
def test_search_documents_via_llama_index_tools_mcp(
    framework_mcp_session: Any,  # ensures seeded corpus + binary on PATH
    sentinel_query: str,
) -> None:
    """FRAME-03: LlamaIndex BasicMCPClient + McpToolSpec -> search_documents."""
    if not HAS_LLAMA_INDEX_MCP:
        pytest.importorskip("llama_index.tools.mcp")

    binary = shutil.which("agent-brain-mcp")
    assert binary is not None, "agent-brain-mcp must be on PATH"

    async def _run() -> list[Any]:
        mcp_client = BasicMCPClient(
            command_or_url=binary,
            args=["--transport", "stdio"],
            env=filtered_subprocess_env(),
        )
        tool_spec = McpToolSpec(client=mcp_client)
        # to_tool_list_async() spawns the stdio subprocess, negotiates,
        # returns a list of LlamaIndex FunctionTool wrappers.
        tools = await tool_spec.to_tool_list_async()
        search_tool = next(
            (t for t in tools if t.metadata.name == "search_documents"),
            None,
        )
        assert search_tool is not None, (
            "search_documents tool not found in LlamaIndex tool list; "
            "available: " + ", ".join(t.metadata.name for t in tools)
        )
        # LlamaIndex tools may use .acall() or .call() depending on
        # version; .acall() is the async path.
        result = await search_tool.acall(
            query=sentinel_query, top_k=3
        )
        return _extract_chunks(result)

    chunks = asyncio.run(_run())
    assert len(chunks) > 0, (
        f"FRAME-03 llamaindex: search_documents({sentinel_query!r}) "
        f"returned 0 chunks; expected >=1 hit on the canned corpus."
    )
```
