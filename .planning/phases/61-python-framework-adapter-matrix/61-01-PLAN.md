def _seed_corpus_via_asyncio_stdio_client(
    binary_path: str, corpus_file_path: str
) -> None:
    """Path B: backend.call_tool does NOT exist — wire stdio_client directly.

    Spawns a fresh stdio client against the same binary the backend
    will spawn, calls add_documents through the SDK's ClientSession,
    then closes. The session-scoped state dir set above (via
    AGENT_BRAIN_STATE_DIR) means the seed lands in the same place the
    backend's later query calls will read from.

    This branch executes only when Task 3 step 0 grep showed
    ``McpStdioBackend.call_tool`` does NOT exist on the class.
    """
    from mcp import ClientSession
    from mcp.client.stdio import StdioServerParameters, stdio_client

    async def _async_seed() -> None:
        params = StdioServerParameters(
            command=binary_path,
            args=[],
            env=filtered_subprocess_env(),
            cwd=None,
        )
        async with stdio_client(params) as (read, write):
            async with ClientSession(read, write) as session:
                await session.initialize()
                await session.call_tool(
                    "add_documents",
                    {"paths": [corpus_file_path]},
                )

    asyncio.run(_async_seed())
