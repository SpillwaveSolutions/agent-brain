```python
def _extract_chunks(result: Any) -> list[Any]:
    """Defensive shape extractor — mirrors Plan 61-02 helper.

    LangChain Tool.ainvoke may return a string (JSON-encoded), a list,
    or a structured object; tolerate all three.
    """
    if result is None:
        return []
    if isinstance(result, list):
        return result
    if isinstance(result, str):
        import json

        try:
            parsed = json.loads(result)
        except (ValueError, TypeError):
            return []
        if isinstance(parsed, list):
            return parsed
        if isinstance(parsed, dict):
            for key in ("chunks", "results", "documents"):
                if key in parsed and isinstance(parsed[key], list):
                    return parsed[key]
        return []
    if isinstance(result, dict):
        for key in ("chunks", "results", "documents"):
            if key in result and isinstance(result[key], list):
                return result[key]
    return []


@pytest.mark.framework_matrix
def test_search_documents_via_langchain_mcp_adapters(
    framework_mcp_session: Any,  # ensures seeded corpus + binary on PATH
    sentinel_query: str,
) -> None:
    """FRAME-02: LangChain MultiServerMCPClient -> search_documents."""
    if not HAS_LANGCHAIN_MCP:
        pytest.importorskip("langchain_mcp_adapters")

    binary = shutil.which("agent-brain-mcp")
    assert binary is not None, "agent-brain-mcp must be on PATH"

    async def _run() -> list[Any]:
        client = MultiServerMCPClient(
            {
                "agent-brain": {
                    "command": binary,
                    "args": ["--transport", "stdio"],
                    "transport": "stdio",
                    "env": filtered_subprocess_env(),
                }
            }
        )
        # get_tools() spawns the stdio subprocess, negotiates the MCP
        # session, and returns a list of LangChain Tool objects mapped
        # from the MCP tool registry.
        tools = await client.get_tools()
        search_tool = next(
            (t for t in tools if t.name == "search_documents"),
            None,
        )
        assert search_tool is not None, (
            "search_documents tool not found in LangChain tool list; "
            "available: " + ", ".join(t.name for t in tools)
        )
        result = await search_tool.ainvoke(
            {"query": sentinel_query, "top_k": 3}
        )
        return _extract_chunks(result)

    chunks = asyncio.run(_run())
    assert len(chunks) > 0, (
        f"FRAME-02 langchain: search_documents({sentinel_query!r}) "
        f"returned 0 chunks; expected >=1 hit on the canned corpus."
    )
```
