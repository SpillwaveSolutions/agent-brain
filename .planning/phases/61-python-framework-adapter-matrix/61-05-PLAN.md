```python
def _extract_chunks(result: Any) -> list[Any]:
    """Defensive shape extractor — mirrors Plan 61-02/03/04 helper."""
    if result is None:
        return []
    if isinstance(result, list):
        return result
    if isinstance(result, str):
        import json

        try:
            parsed = json.loads(result)
        except (ValueError, TypeError):
            return []
        if isinstance(parsed, list):
            return parsed
        if isinstance(parsed, dict):
            for key in ("chunks", "results", "documents"):
                if key in parsed and isinstance(parsed[key], list):
                    return parsed[key]
        return []
    if isinstance(result, dict):
        for key in ("chunks", "results", "documents"):
            if key in result and isinstance(result[key], list):
                return result[key]
    # Pydantic AI tool results may wrap content in .content or .output.
    content = getattr(result, "content", None) or getattr(result, "output", None)
    if content is not None:
        return _extract_chunks(content)
    return []


@pytest.mark.framework_matrix
def test_search_documents_via_pydantic_ai_mcp_server_stdio(
    framework_mcp_session: Any,  # ensures seeded corpus + binary on PATH
    sentinel_query: str,
) -> None:
    """FRAME-04: pydantic_ai.mcp.MCPServerStdio -> search_documents."""
    if not HAS_PYDANTIC_AI:
        pytest.importorskip("pydantic_ai.mcp")

    binary = shutil.which("agent-brain-mcp")
    assert binary is not None, "agent-brain-mcp must be on PATH"

    async def _run() -> list[Any]:
        server = MCPServerStdio(
            command=binary,
            args=["--transport", "stdio"],
            env=filtered_subprocess_env(),
        )
        async with server:
            result = await server.call_tool(
                "search_documents",
                {"query": sentinel_query, "top_k": 3},
            )
        return _extract_chunks(result)

    chunks = asyncio.run(_run())
    assert len(chunks) > 0, (
        f"FRAME-04 pydantic_ai: search_documents({sentinel_query!r}) "
        f"returned 0 chunks; expected >=1 hit on the canned corpus."
    )
```
