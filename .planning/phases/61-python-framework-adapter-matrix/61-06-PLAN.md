```python
def _extract_chunks(result: Any) -> list[Any]:
    """Defensive shape extractor — mirrors Plans 61-02..61-05 helper."""
    if result is None:
        return []
    if isinstance(result, list):
        return result
    if isinstance(result, str):
        import json

        try:
            parsed = json.loads(result)
        except (ValueError, TypeError):
            return []
        if isinstance(parsed, list):
            return parsed
        if isinstance(parsed, dict):
            for key in ("chunks", "results", "documents"):
                if key in parsed and isinstance(parsed[key], list):
                    return parsed[key]
        return []
    if isinstance(result, dict):
        for key in ("chunks", "results", "documents"):
            if key in result and isinstance(result[key], list):
                return result[key]
    # Autogen workbench results may wrap content in .result, .content,
    # or .output depending on version.
    for attr in ("result", "content", "output"):
        nested = getattr(result, attr, None)
        if nested is not None:
            return _extract_chunks(nested)
    return []


@pytest.mark.framework_matrix
def test_search_documents_via_autogen_mcp_workbench(
    framework_mcp_session: Any,  # ensures seeded corpus + binary on PATH
    sentinel_query: str,
) -> None:
    """FRAME-05: autogen_ext.tools.mcp.McpWorkbench -> search_documents."""
    if not HAS_AUTOGEN_MCP:
        pytest.importorskip("autogen_ext.tools.mcp")

    binary = shutil.which("agent-brain-mcp")
    assert binary is not None, "agent-brain-mcp must be on PATH"

    async def _run() -> list[Any]:
        params = StdioServerParams(
            command=binary,
            args=["--transport", "stdio"],
            env=filtered_subprocess_env(),
        )
        workbench = McpWorkbench(server_params=params)
        async with workbench:
            result = await workbench.call_tool(
                name="search_documents",
                arguments={"query": sentinel_query, "top_k": 3},
            )
        return _extract_chunks(result)

    chunks = asyncio.run(_run())
    assert len(chunks) > 0, (
        f"FRAME-05 autogen: search_documents({sentinel_query!r}) "
        f"returned 0 chunks; expected >=1 hit on the canned corpus."
    )
```
