       ```python
       # 2d. Nested storage.postgres.* validation
       storage_data = config.get("storage")
       if isinstance(storage_data, dict):
           pg_data = storage_data.get("postgres")
           if isinstance(pg_data, dict):
               for pg_key in pg_data:
                   if pg_key not in POSTGRES_KNOWN_FIELDS:
                       errors.append(
                           ConfigValidationError(
                               field=f"storage.postgres.{pg_key}",
                               message=f"Unknown key '{pg_key}' in section 'storage.postgres'",
                               line_number=None,
                               suggestion=(
                                   f"Remove '{pg_key}' or check for a typo. "
                                   f"Known fields: {', '.join(sorted(POSTGRES_KNOWN_FIELDS))}"
                               ),
                           )
                       )
               # Type checks for postgres int fields
               for int_key, int_msg in POSTGRES_INT_FIELDS.items():
                   val = pg_data.get(int_key)
                   if val is not None and not isinstance(val, int):
                       errors.append(
                           ConfigValidationError(
                               field=f"storage.postgres.{int_key}",
                               message=int_msg,
                               line_number=None,
                               suggestion=f"Change '{int_key}' to an integer value.",
                           )
                       )
               # Type checks for postgres bool fields
               for bool_key, bool_msg in POSTGRES_BOOL_FIELDS.items():
                   val = pg_data.get(bool_key)
                   if val is not None and not isinstance(val, bool):
                       errors.append(
                           ConfigValidationError(
                               field=f"storage.postgres.{bool_key}",
                               message=bool_msg,
                               line_number=None,
                               suggestion=f"Change '{bool_key}' to a boolean value.",
                           )
                       )
       ```
