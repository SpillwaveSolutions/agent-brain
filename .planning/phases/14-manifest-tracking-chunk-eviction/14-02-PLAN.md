When there are unchanged files (incremental), BM25 needs ALL nodes. After the existing BM25 rebuild call, add:

# For incremental runs, BM25 must include unchanged file chunks too
if (eviction_summary is not None
    and eviction_summary.files_unchanged
    and self.manifest_tracker is not None
    and prior_manifest is not None):
    # Collect unchanged chunk IDs from prior manifest
    unchanged_ids = []
    for fp in eviction_summary.files_unchanged:
        if fp in prior_manifest.files:
            unchanged_ids.extend(prior_manifest.files[fp].chunk_ids)
    if unchanged_ids:
        # Fetch unchanged chunks from storage and combine with new for BM25
        unchanged_nodes = []
        for chunk_id in unchanged_ids:
            try:
                result = await self.storage_backend.get_by_id(chunk_id)
                if result:
                    node = TextNode(
                        id_=chunk_id,
                        text=result.get("text", ""),
                        metadata=result.get("metadata", {}),
                    )
                    unchanged_nodes.append(node)
            except Exception as e:
                logger.warning(f"Could not fetch chunk {chunk_id} for BM25: {e}")
        # Combine new + unchanged nodes for full BM25 rebuild
        all_bm25_nodes = list(all_text_nodes) + unchanged_nodes
        if hasattr(self.storage_backend, 'bm25_manager'):
            bm25 = self.storage_backend.bm25_manager
            if bm25:
                bm25.build_index(all_bm25_nodes)
                logger.info(f"BM25 rebuilt with {len(all_bm25_nodes)} total nodes (incremental)")
