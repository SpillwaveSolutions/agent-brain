       ```python
       @config_group.command("wizard")
       def wizard() -> None:
           """Interactive configuration wizard for Agent Brain providers."""
       ```

    3. Wizard flow:
       a. Prompt for embedding provider: `click.prompt("Embedding provider", type=click.Choice(["openai", "ollama", "cohere"]), default="openai")`
       b. Prompt for embedding model: `click.prompt("Embedding model", default=<provider-specific default>)`
          - openai default: `"text-embedding-3-large"`
          - ollama default: `"nomic-embed-text"`
          - cohere default: `"embed-english-v3.0"`
       c. **If provider == "ollama" ONLY** (per LOCKED decision — skip for openai/cohere):
          - Prompt batch_size: `click.prompt("Batch size", type=click.Choice(["1", "5", "10", "20", "50", "100"]), default="10")`
          - Prompt request_delay_ms: `click.prompt("Request delay between batches (ms, 0=none)", type=click.IntRange(min=0), default=0)`
          - `max_retries` is NOT prompted (per LOCKED decision)
       d. Prompt for summarization provider: `click.prompt("Summarization provider", type=click.Choice(["anthropic", "openai", "ollama", "gemini"]), default="anthropic")`
       e. Prompt for summarization model with provider-specific defaults.

    4. Build config dict from answers:
       ```python
       config: dict[str, Any] = {
           "embedding": {
               "provider": embed_provider,
               "model": embed_model,
           },
           "summarization": {
               "provider": summ_provider,
               "model": summ_model,
           },
       }
       if embed_provider == "ollama":
           config["embedding"]["params"] = {
               "batch_size": int(batch_size),
               "request_delay_ms": request_delay_ms,
           }
       ```

    5. Determine config output path:
       - Look for existing `.agent-brain/` directory by walking up from CWD (same as `_find_config_file` logic)
       - If found, write to `.agent-brain/config.yaml`
       - If not found, create `.agent-brain/config.yaml` in CWD
       - Use `yaml.safe_dump(config, f, default_flow_style=False)`

    6. Print confirmation: `console.print(f"[green]Config written to {config_path}[/]")`

    7. Validate `request_delay_ms` input: `click.IntRange(min=0)` handles rejection of negative values automatically. For `batch_size`, `click.Choice` handles rejection of invalid values (e.g., "7").

    ```python
    """Integration tests for agent-brain config wizard command."""

    import os
    from pathlib import Path

    import yaml
    from click.testing import CliRunner

    from agent_brain_cli.commands.config import config_group


    class TestConfigWizard:
        """Tests for config wizard sub-command."""

        def test_wizard_shows_ollama_prompts(self, tmp_path: Path) -> None:
            """Wizard shows batch_size and request_delay_ms when provider is ollama."""
            runner = CliRunner()
            with runner.isolated_filesystem(temp_dir=tmp_path):
                # Input: ollama, nomic-embed-text, batch_size=20, delay=100, anthropic, claude model
                result = runner.invoke(
                    config_group,
                    ["wizard"],
                    input="ollama\nnomic-embed-text\n20\n100\nanthropic\nclaude-haiku-4-5-20251001\n",
                )
                assert result.exit_code == 0, result.output
                assert "Batch size" in result.output
                assert "Request delay" in result.output
                # Verify config written with ollama params
                config_file = Path(".agent-brain") / "config.yaml"
                assert config_file.exists()
                config = yaml.safe_load(config_file.read_text())
                assert config["embedding"]["params"]["batch_size"] == 20
                assert config["embedding"]["params"]["request_delay_ms"] == 100

        def test_wizard_skips_prompts_for_openai_and_cohere(self, tmp_path: Path) -> None:
            """Wizard does NOT show batch_size/request_delay_ms for openai or cohere."""
            runner = CliRunner()
            for provider in ["openai", "cohere"]:
                with runner.isolated_filesystem(temp_dir=tmp_path):
                    # Input: provider, model, summarization provider, summarization model
                    result = runner.invoke(
                        config_group,
                        ["wizard"],
                        input=f"{provider}\nsome-model\nanthropic\nclaude-haiku-4-5-20251001\n",
                    )
                    assert result.exit_code == 0, f"{provider}: {result.output}"
                    assert "Batch size" not in result.output, f"{provider} should not show Batch size"
                    assert "Request delay" not in result.output, f"{provider} should not show Request delay"

        def test_wizard_rejects_invalid_batch_size(self, tmp_path: Path) -> None:
            """Wizard rejects batch_size values not in [1, 5, 10, 20, 50, 100]."""
            runner = CliRunner()
            with runner.isolated_filesystem(temp_dir=tmp_path):
                # Input: ollama, model, invalid batch_size "7", then valid "10", delay, summ provider, summ model
                result = runner.invoke(
                    config_group,
                    ["wizard"],
                    input="ollama\nnomic-embed-text\n7\n10\n0\nanthropic\nclaude-haiku-4-5-20251001\n",
                )
                assert result.exit_code == 0, result.output
                # Click.Choice shows an error for invalid value and re-prompts
                assert "invalid choice" in result.output.lower() or "7" in result.output

        def test_wizard_rejects_negative_request_delay(self, tmp_path: Path) -> None:
            """Wizard rejects negative request_delay_ms values."""
            runner = CliRunner()
            with runner.isolated_filesystem(temp_dir=tmp_path):
                # Input: ollama, model, batch_size=10, negative delay -1, then valid 0, summ provider, summ model
                result = runner.invoke(
                    config_group,
                    ["wizard"],
                    input="ollama\nnomic-embed-text\n10\n-1\n0\nanthropic\nclaude-haiku-4-5-20251001\n",
                )
                assert result.exit_code == 0, result.output
                # IntRange(min=0) shows error for negative values
                config_file = Path(".agent-brain") / "config.yaml"
                if config_file.exists():
                    config = yaml.safe_load(config_file.read_text())
                    assert config["embedding"]["params"]["request_delay_ms"] >= 0
    ```
