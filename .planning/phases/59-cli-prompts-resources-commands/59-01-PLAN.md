```python
def open_mcp_backend(
    ctx: click.Context, *, timeout: float = 30.0
) -> McpBackend:
    """Construct an McpBackend for MCP-only CLI commands (Phase 59).

    Enforces --transport mcp as the SINGLE point of contract — every MCP-only
    command (agent-brain prompt, agent-brain resources *) calls this instead
    of open_backend, so the transport check lives in one place. Mirrors the
    Phase 57 §3.5 no-silent-fallback contract.

    Raises:
        click.UsageError (exit code 2): when ctx.obj['transport_hint'] is
            not 'mcp', with verbatim wording per CONTEXT.md.
    """
    obj = ctx.obj or {}
    transport_hint = (obj.get("transport_hint") or "").lower()
    if transport_hint != "mcp":
        # Wording verbatim from CONTEXT.md §decisions (--transport mcp requirement)
        # The {cmd_label} is filled by the caller via a thin wrapper if needed,
        # but the open_mcp_backend default error is sufficient for v3.
        raise click.UsageError(
            "This command requires --transport mcp; example: "
            "agent-brain --transport mcp --mcp-transport stdio <command>"
        )

    # Dispatch via the EXISTING open_backend logic but ONLY return the MCP
    # backend half — the function below shares the open_backend dispatch
    # state_dir + mcp_target resolution, then casts to McpBackend.
    state_dir = _resolve_state_dir_for_discovery()
    mcp_transport, mcp_target = resolve_mcp_transport(
        mcp_transport_hint=obj.get("mcp_transport_hint"),
        mcp_url_override=obj.get("mcp_url_override"),
        state_dir=state_dir,
    )
    try:
        from agent_brain_mcp.client import McpHttpBackend, McpStdioBackend
    except ImportError as exc:
        raise click.UsageError(
            "install agent-brain-mcp to use --transport mcp"
        ) from exc

    if mcp_transport == "stdio":
        if shutil.which("agent-brain-mcp") is None:
            raise click.UsageError(
                "agent-brain-mcp not found on PATH; install "
                "agent-brain-mcp into the same Python environment"
            )
        return cast(McpBackend, McpStdioBackend(command="agent-brain-mcp"))
    assert mcp_target is not None
    return cast(McpBackend, McpHttpBackend(url=mcp_target, timeout=timeout))
```
