```python
async def _async_get_prompt(
    self, name: str, arguments: dict[str, str] | None
) -> dict[str, Any]:
    from mcp import ClientSession
    from mcp.client.stdio import stdio_client

    async with stdio_client(self._stdio_params()) as (read, write):
        async with ClientSession(read, write) as session:
            await session.initialize()
            result = await session.get_prompt(name, arguments)
    # result is a GetPromptResult — model_dump to dict[str, Any].
    return result.model_dump(mode="json", exclude_none=False)
```

async def _async_get_prompt(
    self, name: str, arguments: dict[str, str] | None
) -> dict[str, Any]:
    from mcp import ClientSession
    from mcp.client.streamable_http import streamablehttp_client

    async with streamablehttp_client(self.url) as (read, write, *_):

```python
"""`agent-brain prompt <name>` — invoke an MCP prompt and print the expansion.

Phase 59 Plan 02 (CLI-MCP-05). Requires --transport mcp (enforced by
open_mcp_backend from Phase 59 Plan 01). Mirrors the no-silent-fallback
contract from v10.2 HTTP-03 and Phase 57 §3.5.

Examples:
    agent-brain --transport mcp --mcp-transport stdio prompt find-callers \\
        --arg symbol=parse_query --arg file=query_service.py
    agent-brain --transport mcp --mcp-transport http prompt explain-architecture
    agent-brain --transport mcp --mcp-transport stdio prompt onboard-to-codebase --json
"""

from __future__ import annotations

import json
import sys

import click
from mcp import McpError

from agent_brain_cli.client.transport import open_mcp_backend


def _parse_arg(arg: str) -> tuple[str, str]:
    """Split --arg KEY=VALUE on the FIRST '='. VALUE may contain '='.

    Raises:
        click.UsageError: when '=' is missing or KEY is empty.
    """
    if "=" not in arg:
        raise click.UsageError(
            f"--arg value must be KEY=VALUE; got {arg!r} (missing '=')"
        )
    key, _, value = arg.partition("=")
    if not key:
        raise click.UsageError(
            f"--arg KEY must be non-empty; got {arg!r}"
        )
    return key, value


@click.command("prompt")
@click.argument("name", required=True)
@click.option(
    "--arg",
    "args",
    multiple=True,
    metavar="KEY=VALUE",
    help="Prompt argument as KEY=VALUE. Repeat for multiple args.",
)
@click.option(
    "--json",
    "as_json",
    is_flag=True,
    default=False,
    help="Print the raw prompts/get response as pretty JSON.",
)
@click.pass_context
def prompt_command(
    ctx: click.Context, name: str, args: tuple[str, ...], as_json: bool
) -> None:
    """Invoke an MCP prompt by name and print its expanded content.

    Requires --transport mcp explicitly; no silent fallback to UDS/HTTP.
    """
    parsed_args: dict[str, str] = {}
    for raw in args:
        key, value = _parse_arg(raw)
        parsed_args[key] = value

    backend = open_mcp_backend(ctx)
    try:
        with backend:  # context-manager surface — close on exit
            try:
                result = backend.get_prompt(name, parsed_args or None)
            except McpError as exc:
                # Unknown prompt name → query list_prompts and re-raise with
                # the available-names list (CONTEXT decision; exit 2).
                try:
                    available_raw = backend.list_prompts()
                except Exception:  # noqa: BLE001 — defensive; never bury original
                    raise click.UsageError(
                        f"Prompt call failed: {exc}"
                    ) from exc
                names = sorted(
                    p.get("name", "") for p in available_raw if p.get("name")
                )
                joined = ", ".join(names) if names else "<no prompts registered>"
                raise click.UsageError(
                    f"Unknown prompt {name!r}; available: {joined}"
                ) from exc
    except click.UsageError:
        raise
    except Exception as exc:  # noqa: BLE001 — surface MCP/SDK failures cleanly
        click.echo(f"Error invoking prompt {name!r}: {exc}", err=True)
        sys.exit(1)

    if as_json:
        click.echo(json.dumps(result, indent=2))
        return

    # Default: render expanded prompt text — concatenate
    # messages[].content.text joined with '\n---\n' between messages.
    messages = result.get("messages", []) or []
    if not messages:
        # No messages — still print the description for context.
        description = result.get("description") or ""
        click.echo(description)
        return
    rendered: list[str] = []
    for message in messages:
        content = message.get("content", {}) or {}
        text = content.get("text", "") or ""
        rendered.append(text)
    click.echo("\n---\n".join(rendered))


__all__ = ["prompt_command"]
```
